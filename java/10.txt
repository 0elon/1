import java.sql.*;
import java.util.Scanner;

public class JDBCDemo {
    static final String URL = "jdbc:mysql://localhost:3306/demo";
    static final String USER = "root";
    static final String PASS = "your_mysql_password";

    public static void main(String[] args) {
        try { Class.forName("com.mysql.cj.jdbc.Driver"); }
        catch (Exception e) { System.out.println("Driver not found"); return; }

        Scanner sc = new Scanner(System.in);
        while (true) {
            System.out.println("\n1.Insert  2.Read  3.Update  4.Delete  5.Exit");
            int ch = sc.nextInt();
            try (Connection con = DriverManager.getConnection(URL, USER, PASS)) {
                switch (ch) {
                    case 1:
                        PreparedStatement ins = con.prepareStatement("INSERT INTO students VALUES(?,?,?)");
                        System.out.print("Enter ID Name Age: ");
                        ins.setInt(1, sc.nextInt());
                        ins.setString(2, sc.next());
                        ins.setInt(3, sc.nextInt());
                        ins.executeUpdate();
                        System.out.println("Inserted!");
                        break;

                    case 2:
                        ResultSet rs = con.createStatement().executeQuery("SELECT * FROM students");
                        while (rs.next())
                            System.out.println(rs.getInt(1) + " " + rs.getString(2) + " " + rs.getInt(3));
                        break;

                    case 3:
                        PreparedStatement upd = con.prepareStatement("UPDATE students SET name=?, age=? WHERE id=?");
                        System.out.print("Enter NewName NewAge ID: ");
                        upd.setString(1, sc.next());
                        upd.setInt(2, sc.nextInt());
                        upd.setInt(3, sc.nextInt());
                        upd.executeUpdate();
                        System.out.println("Updated!");
                        break;

                    case 4:
                        PreparedStatement del = con.prepareStatement("DELETE FROM students WHERE id=?");
                        System.out.print("Enter ID to delete: ");
                        del.setInt(1, sc.nextInt());
                        del.executeUpdate();
                        System.out.println("Deleted!");
                        break;

                    case 5: return;
                }
            } catch (Exception e) { System.out.println("Error: " + e); }
        }
    }
}
---

- Install MySQL Server.
Create database demo.
Create table:
CREATE TABLE students (
  id INT PRIMARY KEY,
  name VARCHAR(50),
  age INT
);
- Download MySQL Connector/J (JAR file) from MySQL official site.
- Create Java Project in Eclipse.
File → New → Java Project.
Name it JDBCDemo.
- Add JDBC JAR to Project.
Right-click project → Build Path → Configure Build Path.
Open Libraries tab → Add External JARs.
Select mysql-connector-j-x.x.x.jar.
- Create Java Class.
Right-click src → New → Class.
Name it JDBCDemo.
Paste the program.
- Update Credentials.
Change USER and PASS in code to match MySQL login.
- Run Program.
Right‑click file → Run As → Java Application.
Use menu:
1 → Insert
2 → Read
3 → Update
4 → Delete
5 → Exit
