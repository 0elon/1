class p1 {
    static void quad(Scanner s){
        double a=s.nextDouble(),b=s.nextDouble(),c=s.nextDouble(),d=b*b-4*a*c;
        if(d>0) System.out.printf("Roots: %.2f %.2f\n",(-b+Math.sqrt(d))/(2*a),(-b-Math.sqrt(d))/(2*a));
        else if(d==0) System.out.printf("Equal Root: %.2f\n",-b/(2*a));
        else System.out.printf("Roots: %.2f Â± %.2fi\n",-b/(2*a),Math.sqrt(-d)/(2*a));
    }
    static void mul(Scanner s){
        int n=s.nextInt();int[]a=new int[n],b=new int[n];
        for(int i=0;i<n;i++)a[i]=s.nextInt();
        for(int i=0;i<n;i++)b[i]=s.nextInt();
        for(int i=0;i<n;i++)System.out.print(a[i]*b[i]+" ");
        System.out.println();
    }
    static void sort(Scanner s){
        int n=s.nextInt();int[]x=new int[n];
        for(int i=0;i<n;i++)x[i]=s.nextInt();
        for(int i=0;i<n;i++)
            for(int j=0;j<n-i-1;j++)
                if(x[j]>x[j+1]){int t=x[j];x[j]=x[j+1];x[j+1]=t;}
        System.out.println("Asc: "+Arrays.toString(x));
        for(int i=0;i<n/2;i++){int t=x[i];x[i]=x[n-i-1];x[n-i-1]=t;}
        System.out.println("Desc: "+Arrays.toString(x));
    }
    public static void main(String[]a){
        Scanner s=new Scanner(System.in);
        for(int c;(c=s.nextInt())!=4;)
            if(c==1)quad(s); else if(c==2)mul(s); else sort(s);
    }
}

class complex{
        double r,i;
        complex(double r,double i){
                this.r=r;this.i=i;
        }
        complex add(complex c){
                return new complex(r+c.r,i+c.i);
        }
        complex sub(complex c){
                return new complex(r-c.r,i-c.i);
        }
        boolean eq(complex c){
                return r==c.r&&i==c.i;
        }
        void show(){
                System.out.println(r+(i>=0?"+":"-")+Math.abs(i)+"i");
        }
}

abstract class Bank{

        double b;
        Bank(String n,int a,double b){this.n=n;this.a=a;this.b=b;}

}

class Shared{
        int x;
        boolean f=false;
        synchronized void prod(int n)throws Exception{
                while(f) wait();
                x=n;
                f=true;
                System.out.println("Produced: "+x);
                notify();
        }
        synchronized void cons(int n)throws Exception{
                while(!f) wait();
                System.out.println("Consumed: "+x);
                f=false;
                notify();
        }
}

class p7{
        public static void main(String[] a){
                Shared s = new Shared();
                new Thread(()->{
                        try{
                                for(int i=1;i<=5;i++)
                                        s.prod(i);
                        }catch(Exception e){}
                }).start();

                new Thread(()->{
                        try{
                                for(int i=1;i<=5;i++)
                                        s.cons(i);
                        }catch(Exception e){}
                }).start();
        }
}

import java.sql.*; import java.util.*;
public class JDBCMini {
    static final String URL="jdbc:mysql://localhost:3306/demo";
    static final String USER="root", PASS="password";

    public static void main(String[] args)throws Exception{
        Scanner sc=new Scanner(System.in);
        Connection c=DriverManager.getConnection(URL,USER,PASS);
        while(true){
            System.out.println("1.Insert 2.Read 3.Update 4.Delete 5.Exit");
            switch(sc.nextInt()){
                case 1: // INSERT
                    PreparedStatement p=c.prepareStatement("INSERT INTO students VALUES(?,?,?)");
                    System.out.print("ID: "); p.setInt(1,sc.nextInt()); sc.nextLine();
                    System.out.print("Name: "); p.setString(2,sc.nextLine());
                    System.out.print("Age: "); p.setInt(3,sc.nextInt());
                    System.out.println(p.executeUpdate()+" row inserted"); break;
                case 2: // READ
                    ResultSet rs=c.createStatement().executeQuery("SELECT * FROM students");
                    while(rs.next()) System.out.println(rs.getInt(1)+" "+rs.getString(2)+" "+rs.getInt(3));
                    break;
                case 3: // UPDATE
                    PreparedStatement u=c.prepareStatement("UPDATE students SET name=?,age=? WHERE id=?");
                    System.out.print("ID: "); int id=sc.nextInt(); sc.nextLine();
                    System.out.print("New Name: "); String n=sc.nextLine();
                    System.out.print("New Age: "); int a=sc.nextInt();
                    u.setString(1,n); u.setInt(2,a); u.setInt(3,id);
                    System.out.println(u.executeUpdate()+" row updated"); break;
                case 4: // DELETE
                    PreparedStatement d=c.prepareStatement("DELETE FROM students WHERE id=?");
                    System.out.print("ID: "); d.setInt(1,sc.nextInt());
                    System.out.println(d.executeUpdate()+" row deleted"); break;
                case 5: System.exit(0);
            }
        }
    }
}
