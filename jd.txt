class Complex {
    private double r,i;
    Complex(double r,double i){this.r=r;this.i=i;}
    Complex add(Complex c){return new Complex(r+c.r,i+c.i);}
    Complex sub(Complex c){return new Complex(r-c.r,i-c.i);}
    boolean eq(Complex c){return r==c.r && i==c.i;}
    void show(){System.out.println(r+(i>=0?"+":"-")+Math.abs(i)+"i");}
}

     }
}

class Shared{
        int x;
        boolean f=false;
        synchronized void prod(int n)throws Exception{
                while(f) wait();
                x=n;
                f=true;
                System.out.println("Produced: "+x);
                notify();
        }
        synchronized void cons(int n)throws Exception{
                while(!f) wait();
                System.out.println("Consumed: "+x);
                f=false;
                notify();
        }
}
class p7{
        public static void main(String[] a){
                Shared s = new Shared();
                new Thread(()->{
                        try{
                                for(int i=1;i<=5;i++)
                                        s.prod(i);
                        }catch(Exception e){}
                }).start();

                new Thread(()->{
                        try{
                                for(int i=1;i<=5;i++)
                                        s.cons(i);
                        }catch(Exception e){}
                }).start();
        }
}
