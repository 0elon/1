//Step1 : Kali linux vm send a UDP broadcast message

import socket
import time
sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
sock.setsockopt(socket.SOL_SOCKET, socket.SO_BROADCAST, 1)
for i in range(5):
msg = &quot;DEVICE_DISCOVERY_{}&quot;.format(i).encode()
sock.sendto(msg, (&quot;255.255.255.255&quot;, 4000))
print(&quot;[Sent]&quot;, msg)
time.sleep(1)

//Step 2: ubuntu vm Listen for broadcast traffic

import socket
sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
sock.bind((&quot;&quot;, 4000))
print(&quot;Listening for broadcast packets...&quot;)
try:
while True:
data, addr = sock.recvfrom(1024)
print(f&quot;Broadcast from {addr}: {data.decode()}&quot;)
except KeyboardInterrupt:
print(“Listener Stopped Manually”)

finally:
sock.close()
